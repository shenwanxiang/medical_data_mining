def get_hosp_D(lng_a,lat_a,lng_b,lat_b):
    import re
    import sys
    import os
    from itertools import islice
    from math import radians,cos,sin,asin,sqrt
    AVG_EARTH_RADIUS = 6371  #in km
    lng1,lat1,lng2,lat2 = map(radians,[lng_a,lat_a,lng_b,lat_b])
    dlng = lng2 - lng1
    dlat = lat2 - lat1
    d = sin(dlat/2)**2 + cos(lat1) * cos(lat2) * sin(dlng/2)**2
    h = 2 * asin(sqrt(d)) * AVG_EARTH_RADIUS
    return h


def get_euclidean_D(lng_a,lat_a,lng_b,lat_b):
    import numpy as np
    euclidean_D = pow((lng_b - lng_a)**2 + (lat_b - lat_a)**2,0.5)
    return euclidean_D

def from_lng_lat_to_xy(hx,hy,lng_array,lat_array):    
    '''
    输入参考点的经纬度hx,hy，计算其他医疗机构的相对位置，单位是km，
    医疗机构的经纬度分别输入，类型是array
    返回医疗机构的相对位置cx，cy，类型是array
    '''
    cx = []
    cy = []
    for x,y in zip(lng_array,lat_array):
        cx.append(get_hosp_D(hx,hy,x,hy))
        cy.append(get_hosp_D(hx,hy,hx,y))
    return np.array(cx),np.array(cy)


def from_fixdot_lnglat_to_xy(home_address, lng_array,lat_array):
    hx = float(home_address.split(':')[0])
    hy = float(home_address.split(':')[1])
    cx = []
    cy = []
    for x,y in zip(lng_array,lat_array):
        if x < hx:
            cx.append(-get_hosp_D(hx,hy,x,hy))
        else: cx.append(get_hosp_D(hx,hy,x,hy))
        if y < hy:
            cy.append(-get_hosp_D(hx,hy,hx,y))
        else: cy.append(get_hosp_D(hx,hy,hx,y))
    return np.array(cx),np.array(cy)

def plot_hosp_times(hosp):
    plt.figure(figsize=(9,6))
    hosp_times = Counter(hosp)
    sort_hosp_times = sorted(hosp_times.items(), key=lambda x: x[1],reverse = True)
    idx = np.arange(len(sort_hosp_times))
    tt = np.array([x[1] for x in sort_hosp_times])
    n = [x[0] for x in sort_hosp_times]
    width = 0.5
    plt.rc('font', family='SimHei', size=13)
    plt.bar(idx, tt, width, color='b',label = '')
    for times,idxs in zip(tt,idx):
        plt.text(idxs, times+ 0.1, times, ha='center', va='bottom')
    plt.xlabel('医院')
    plt.ylabel('次数')
    plt.xticks(idx, n, rotation=90)

    plt.show()
